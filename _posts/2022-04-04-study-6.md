```cpp
#include <iostream>
using namespace std;
const int MAX = 10;
int* dyIntArr(int n);

int main() {

	int A[MAX] = { 999 }; // static 메모리 allocation
	cout << "A : " << A << " *(A+0) : " << *(A + 0) << endl;
	cout << "&A[0] : " << &A[0] << endl;
	int size = 0;
	
	cout << "&size : " << &size << endl;
	cout << "how many elements? : ";
	cin >> size;

	// int* B = new int[size]; // dynamic memory allocation, on heap
	int* B = dyIntArr(size); // 얘가 41라인 받음

	for (int i = 0; i < size; i++) {

		cout << "입력 : ";
		cin >> *(B + i);
	}
	cout << "&B : " << &B << endl;
	cout << "B : " << B << " *(B+0) : " << *(B + 0) << endl;
	// 합을 구해서 평균을 계산한 후 출력한다.
	int  total = 0;
	for (int i = 0; i < size; i++) {
		total += *(B + i);
	}

	cout << "총합 : " << total << endl;
	cout << "평균 : " << total / size << endl;
	delete[] B;
	B = NULL;

	return 0;
}
```
