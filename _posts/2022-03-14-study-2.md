### cin cout 예제 (SimpleInput)

```cpp
#include <iostream>

using namespace std;

int main() {

	cout << "Width : ";
	int w;
	cin >> w;

	cout << "Height : ";
	int h;
	cin >> h;

	int area = w * h;
	cout << "면적은 " << area << " 입니다." << endl; 

	return 0;
}
```

### Password 예제

<br>

```cpp
#include <iostream>

using namespace std;

int main() {

	cout << "암호를 입력하세요 : " << endl;
	char pwd[100];

	while (true) {

		cout << "암호 : ";
		cin >> pwd;
		if (strcmp(pwd,"C++") == 0) {
			cout << "프로그램을 정상 종료합니다." << endl;
			break;

		}

		else {
			cout << "암호가 틀립니다! " << endl;

		}

	}

	return 0;

}
```
